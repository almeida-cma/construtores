public class Car {

    private String color;
    private int year;

    // Construtor padr√£o
    public Car() {
        this.color = "Unknown";
        this.year = 0;
    }

    // Construtor sobrecarregado
    public Car(String color, int year) {
        this.color = color;
        this.year = year;
    }

    // Getter para color
    public String getColor() {
        return this.color;
    }

    // Getter para year
    public int getYear() {
        return this.year;
    }
}
